			if (i % uzunluk == 0 || i % uzunluk == uzunluk -1) {
				if (j % uzunluk != 0)
					printf("* ");
				else 
					printf("*");
			}else {
				if (j % uzunluk == 1){
					printf("* ");

				}else if (j % uzunluk == 0){
					printf("*");

				}else
					printf("  ");
			}
		}
		printf("\n");
	}
}
