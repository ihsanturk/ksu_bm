# include <stdio.h>
int main() {
	int i=0, j=0, uzunluk=0;

	do {	
		printf("Ici dolu kare cizdirmek icin 3-15 arasinda bir sayi giriniz.");
		scanf("%d", &uzunluk);
	
	}while(uzunluk < 3 || uzunluk > 15);
	
	for (i=1; i < (uzunluk+1); i++){
		for (j=1; j < (uzunluk+1); j++){
			if (j % uzunluk != 0)
				printf("* ");
			else
				printf("*");
		}
		printf("\n");
	}
}
