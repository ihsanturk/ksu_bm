# include <stdio.h>

	int i=1, maksimum, sayi;

	while(i < 4){
		printf("%d. sayiyi giriniz: ", i);
		scanf("%d", &sayi);
		
		if (i == 1) maksimum = sayi;

		i++;
	}

}
